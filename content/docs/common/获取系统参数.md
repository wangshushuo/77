```ts
const key = 'timesheetCostRateType'; // 参数名
const settingDataLoader = new DataLoader(EN_Setting, ['values.value'], {
  criteriaStr: `key='${key}'`,
} as QueryOptions);
const data = await settingDataLoader.query();
const result = data[0]; // result.value就是该参数的值
```
